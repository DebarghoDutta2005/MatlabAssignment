
x = [1, 2, 3, 4, 5];
folded_x = fliplr(x); 
shift_amount = 2; 
shifted_x = [zeros(1, shift_amount), x]; 

figure;

subplot(3, 1, 1);
stem(x, 'filled');
title('Original Sequence');
xlabel('n');
ylabel('Amplitude');
grid on;

subplot(3, 1, 2);
stem(folded_x, 'filled');
title('Folded Sequence');
xlabel('n');
ylabel('Amplitude');
grid on;

subplot(3, 1, 3);
stem(shifted_x, 'filled');
title('Shifted Sequence (Right by 2)');
xlabel('n');
ylabel('Amplitude');
grid on;
